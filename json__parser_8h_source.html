134 inline std::istream& operator>>(std::istream &in, JsonData*& json) {
135  while (in) {
136  char ch;
137  in >> ch;
138  switch (ch) {
139  case '{': {
140  ordered_map<std::string, JsonData*> obj;
141  do {
142  in >> std::ws >> ch;
143  if (ch == '}')
144  break;
145  in.unget();
146 
147  JsonData *key, *val;
148  in >> key >> std::ws >> ch >> std::ws >> val;
149  obj[*(dynamic_cast<std::string*>(key))] = val;
150 
151  in >> std::ws >> ch;
152  } while (in && ch != '}');
153 
154  json = new JsonObject(obj);
155  return in;
156  }
157  case '[': {
158  std::vector<JsonData*> vec;
159  do {
160  in >> std::ws >> ch;
161  if (ch == ']')
162  break;
163  in.unget();
164 
165  JsonData *elem;
166  in >> elem;
167  vec.push_back(elem);
168 
169  in >> std::ws >> ch;
170  } while (in && ch != ']');
171 
172  json = new JsonVector(vec);
173  return in;
174  }
175  case '"': {
176  std::string s;
177  getline(in, s, '"');
178  json = new JsonString(s);
179  return in;
180  }
181  case '-': case '0': case '1': case '2': case '3':
182  case '4': case '5': case '6': case '7': case '8': case '9': {
183  mpz_class num;
184  in.unget();
185  in >> num;
186  json = new JsonNumber(num);
187  return in;
188  }
189  case 't': case 'T':
190  in.ignore(3);
191  json = new JsonBoolean(true);
192  return in;
193  case 'f': case 'F':
194  in.ignore(4);
195  json = new JsonBoolean(false);
196  return in;
197  case 'n': case 'N':
198  in.ignore(3);
