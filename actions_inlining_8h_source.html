26 struct ActionCallInfo {
27  const IR::P4Action* caller;
28  const IR::P4Action* callee;

31  ActionCallInfo(const IR::P4Action* caller, const IR::P4Action* callee,

34  { CHECK_NULL(caller); CHECK_NULL(callee); CHECK_NULL(call); }

36  { out << callee << " into " << caller << " at " << call; }

39 struct AInlineWorkList {
40  // Map caller -> statement -> callee
41  std::map<const IR::P4Action*,
42  std::map<const IR::MethodCallStatement*, const IR::P4Action*>> sites;
43  void add(ActionCallInfo* info) {
44  CHECK_NULL(info);
45  sites[info->caller][info->call] = info->callee;
46  }
47  void dbprint(std::ostream& out) const;
48  bool empty() const
49  { return sites.empty(); }
50 };

52 class ActionsInlineList {
53  std::vector<ActionCallInfo*> toInline; // initial data
54  std::vector<ActionCallInfo*> inlineOrder; // sorted in inlining order
55  public:
56  void analyze(); // generate the inlining order
57  AInlineWorkList* next();
58  void add(ActionCallInfo* aci)
59  { toInline.push_back(aci); }
60 
61  void replace(const IR::P4Action* container, const IR::P4Action* replacement) {
62  LOG1("Substituting " << container << " with " << replacement);
63  for (auto e : inlineOrder) {
64  if (e->callee == container)
65  e->callee = replacement;
66  if (e->caller == container)
67  e->caller = replacement;
68  }
69  }
70 };

150 // Special inliner which works directly on P4 v1.0 representation
151 class InlineActions : public Transform {
152  const IR::V1Program *global;
153  class SubstActionArgs : public Transform {
154  const IR::ActionFunction *function;
155  const IR::Primitive *callsite;
156  const IR::Node *postorder(IR::ActionArg *arg) override {
157  for (unsigned i = 0; i < function->args.size(); ++i)
158  if (function->args[i] == getOriginal())
159  return callsite->operands[i];
160  BUG("Action arg not argument of action");
161  return arg; }
162  public:
163  SubstActionArgs(const IR::ActionFunction *f, const IR::Primitive *c)
164  : function(f), callsite(c) {}
165  };
166  const IR::V1Program *preorder(IR::V1Program *gl) override { return global = gl; }
167  const IR::Node *preorder(IR::Primitive *p) override {
168  if (auto af = global->get<IR::ActionFunction>(p->name))
169  return af->action.clone()->apply(SubstActionArgs(af, p));
170  return p; }
171 };
