109  // Context access/search functions. getContext returns the context
110  // that refers to the immediate parent of the node currently being
111  // visited. findContext searches up the context for a (grand)parent
112  // of a specific type. Orig versions return the original node (before
113  // any cloning or modifications)
114  const IR::Node* getOriginal() const { return ctxt->original; }
115  template <class T>
116  const T* getOriginal() const {
117  CHECK_NULL(ctxt->original);
118  BUG_CHECK(ctxt->original->is<T>(), "%1% does not have the expected type %2%",
119  ctxt->original, typeid(T).name());
120  return ctxt->original->to<T>();
121  }

128  template <class T> inline const T *findContext(const Context *&c) const {
129  if (!c) c = ctxt;
130  while ((c = c->parent))
131  if (auto *rv = dynamic_cast<const T *>(c->node)) return rv;
132  return nullptr; }
