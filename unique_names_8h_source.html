26 class RenameMap {
27  // Internal declaration name
28  std::map<const IR::IDeclaration*, cstring> newName;
29  std::set<const IR::P4Action*> inTable; // all actions that appear in tables
30 
31  public:
32  void setNewName(const IR::IDeclaration* decl, cstring name) {
33  CHECK_NULL(decl);
34  BUG_CHECK(!name.isNullOrEmpty(), "Empty name");
35  LOG1("Renaming " << decl << " to " << name);
36  if (newName.find(decl) != newName.end())
37  BUG("%1%: already renamed", decl);
38  newName.emplace(decl, name);
39  }
40  cstring getName(const IR::IDeclaration* decl) const {
41  CHECK_NULL(decl);
42  BUG_CHECK(newName.find(decl) != newName.end(), "%1%: no new name", decl);
43  auto result = ::get(newName, decl);
44  return result;
45  }
46  bool toRename(const IR::IDeclaration* decl) const {
47  CHECK_NULL(decl);
48  return newName.find(decl) != newName.end();
49  }
50  void foundInTable(const IR::P4Action* action)
51  { inTable.emplace(action); }
52  bool isInTable(const IR::P4Action* action)
53  { return inTable.find(action) != inTable.end(); }
54 };

114 // Finds parameters for actions that will be given unique names
115 class FindParameters : public Inspector {
116  ReferenceMap* refMap; // used to generate new names
117  RenameMap* renameMap;
118 
119  // If all is true then rename all parameters, else rename only
120  // directional parameters
121  void doParameters(const IR::ParameterList* pl, bool all) {
122  for (auto p : *pl->parameters) {
123  if (!all && p->direction == IR::Direction::None)
124  continue;
125  cstring newName = refMap->newName(p->name);
126  renameMap->setNewName(p, newName);
127  }
128  }
129  public:
130  FindParameters(ReferenceMap* refMap, RenameMap* renameMap) :
131  refMap(refMap), renameMap(renameMap)
132  { CHECK_NULL(refMap); CHECK_NULL(renameMap); setName("FindParameters"); }
133  void postorder(const IR::P4Action* action) override {
134  bool inTable = renameMap->isInTable(action);
135  doParameters(action->parameters, !inTable);
136  }
137 };
