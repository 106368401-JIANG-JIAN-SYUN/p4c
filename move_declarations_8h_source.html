25 // Moves all local declarations to the "top".
26 // This can only be done safely if all declarations have different names,
27 // so it has to be done after the UniqueNames pass.
28 // We move all declarations in a control or parser to the "top", including
29 // the ones in the Control body and Parser states.
30 // Also, if the control has nested actions, we move the declarations from
31 // the actions to the enclosing control.
32 class MoveDeclarations : public Transform {
33  std::vector<IR::Vector<IR::Declaration>*> toMove;
34  void push() { toMove.push_back(new IR::Vector<IR::Declaration>()); }
35  void pop() { BUG_CHECK(!toMove.empty(), "Empty move stack"); toMove.pop_back(); }
36  IR::Vector<IR::Declaration> *getMoves() const
37  { BUG_CHECK(!toMove.empty(), "Empty move stack"); return toMove.back(); }
38  void addMove(const IR::Declaration* decl)
39  { getMoves()->push_back(decl); }
40 
41  public:
42  MoveDeclarations() { setName("MoveDeclarations"); }
43  void end_apply(const IR::Node*) override
44  { BUG_CHECK(toMove.empty(), "Non empty move stack"); }
45  const IR::Node* preorder(IR::P4Action* action) override {
46  if (findContext<IR::P4Control>() == nullptr)
47  // If we are not in a control, move to the beginning of the action.
48  // Otherwise move to the control's beginning.
49  push();
50  return action; }
51  const IR::Node* preorder(IR::P4Control* control) override
52  { push(); return control; }
53  const IR::Node* preorder(IR::P4Parser* parser) override
54  { push(); return parser; }
55  const IR::Node* preorder(IR::Function* function) override
56  { push(); return function; }
57  const IR::Node* postorder(IR::P4Action* action) override;
58  const IR::Node* postorder(IR::P4Control* control) override;
59  const IR::Node* postorder(IR::P4Parser* parser) override;
60  const IR::Node* postorder(IR::Function* function) override;
61  const IR::Node* postorder(IR::Declaration_Variable* decl) override;
62  const IR::Node* postorder(IR::Declaration_Constant* decl) override;
63 };
