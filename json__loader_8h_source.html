62  const IR::Node* get_node() {
63  if (!json || !json->is<JsonObject>()) return nullptr; // invalid json exception?
64  int id = json->to<JsonObject>()->get_id();
65  if (id >= 0) {
66  if (node_refs.find(id) == node_refs.end()) {
67  if (auto fn = get(IR::unpacker_table, json->to<JsonObject>()->get_type()))
68  node_refs[id] = fn(*this);
69  else
70  return nullptr; } // invalid json exception?
71  return node_refs[id]; }
72  return nullptr; // invalid json exception?
73  }
