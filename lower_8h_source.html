63 // If a select contains an expression which
64 // looks too complicated, it is lifted into a temporary.
65 class RemoveExpressionsFromSelects : public Transform {
66  P4::ReferenceMap* refMap;
67  P4::TypeMap* typeMap;
68  IR::IndexedVector<IR::Declaration> newDecls;
69  IR::IndexedVector<IR::StatOrDecl> assignments;
70 
71  public:
72  RemoveExpressionsFromSelects(P4::ReferenceMap* refMap, P4::TypeMap* typeMap):
73  refMap(refMap), typeMap(typeMap)
74  { CHECK_NULL(refMap); CHECK_NULL(typeMap); setName("RemoveExpressionsFromSelects"); }
75  const IR::Node* postorder(IR::SelectExpression* expression) override;
76  const IR::Node* preorder(IR::ParserState* state) override
77  { assignments.clear(); return state; }
78  const IR::Node* postorder(IR::ParserState* state) override {
79  if (!assignments.empty()) {
80  auto comp = new IR::IndexedVector<IR::StatOrDecl>(*state->components);
81  comp->append(assignments);
82  state->components = comp;
83  }
84  return state;
85  }
86  const IR::Node* preorder(IR::P4Parser* parser) override
87  { newDecls.clear(); return parser; }
88  const IR::Node* postorder(IR::P4Parser* parser) override {
89  if (!newDecls.empty()) {
90  auto locals = new IR::IndexedVector<IR::Declaration>(*parser->parserLocals);
91  locals->append(newDecls);
92  parser->parserLocals = locals;
93  }
94  return parser;
95  }
96 };
