28 // Represents compile-time information about a MethodCallExpression
29 class MethodInstance {
30  protected:
31  MethodInstance(const IR::MethodCallExpression* mce,
32  const IR::IDeclaration* decl,
33  const IR::Type_MethodBase* originalMethodType,
34  const IR::Type_MethodBase* actualMethodType) :
35  expr(mce), object(decl), originalMethodType(originalMethodType),
36  actualMethodType(actualMethodType)
37  { CHECK_NULL(mce); CHECK_NULL(originalMethodType); CHECK_NULL(actualMethodType); }
38 
39  public:
40  const IR::MethodCallExpression* expr;
41  const IR::IDeclaration* object; // Declaration of object that method is applied to.
42  // May be null for plain functions.
43  const IR::Type_MethodBase* originalMethodType; // the type of the *original* called method,
44  // without instantiated type parameters.
45  const IR::Type_MethodBase* actualMethodType; // type of called method,
46  // with instantiated type parameters.
47 
48  virtual bool isApply() const { return false; }
49  virtual ~MethodInstance() {}
50 
51  static MethodInstance* resolve(const IR::MethodCallExpression* mce,
52  ReferenceMap* refMap, TypeMap* typeMap,
53  bool useExpressionType = false);
54  static MethodInstance* resolve(const IR::MethodCallStatement* mcs,
55  ReferenceMap* refMap, TypeMap* typeMap)
56  { return resolve(mcs->methodCall, refMap, typeMap); }
57  const IR::ParameterList* getOriginalParameters() const
58  { return originalMethodType->parameters; }
59  const IR::ParameterList* getActualParameters() const
60  { return actualMethodType->parameters; }
61  template<typename T> bool is() const { return to<T>() != nullptr; }
62  template<typename T> const T* to() const { return dynamic_cast<const T*>(this); }
63 };
