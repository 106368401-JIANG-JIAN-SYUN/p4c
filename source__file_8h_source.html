41 /* A character position within some InputSources: a pair of
42  line/column positions. Can only be interpreted in the context
43  of some InputSources.
44 
45  In files line numbering starts at 1, so 0 is reserved for
46  "invalid" source positions. As a consequence, invalid source
47  position are the "smallest", which is a reasonable choice.
48 */
49 class SourcePosition final {
50  public:
51  // Creates an invalid source position
52  SourcePosition()
53  : lineNumber(0),
54  columnNumber(0) {}
55 
56  SourcePosition(unsigned lineNumber, unsigned columnNumber);
57 
58  SourcePosition(const SourcePosition& other)
59  : lineNumber(other.lineNumber),
60  columnNumber(other.columnNumber) {}
61 
62  inline bool operator==(const SourcePosition& rhs) const {
63  return this->columnNumber == rhs.columnNumber &&
64  this->lineNumber == rhs.columnNumber;
65  }
66  inline bool operator!=(const SourcePosition& rhs) const
67  {return !this->operator==(rhs);}
68 
69  inline bool operator< (const SourcePosition& rhs) const {
70  return (this->lineNumber < rhs.lineNumber) ||
71  (this->lineNumber == rhs.lineNumber &&
72  this->columnNumber < rhs.columnNumber);
73  }
74  inline bool operator> (const SourcePosition& rhs) const
75  {return rhs.operator< (*this);}
76  inline bool operator<=(const SourcePosition& rhs) const
77  {return !this->operator> (rhs);}
78  inline bool operator>=(const SourcePosition& rhs) const
79  {return !this->operator< (rhs);}
80 
81  // Move one column back. This never moves one line back.
82  SourcePosition& operator--() {
83  if (this->columnNumber > 0)
84  this->columnNumber--;
85  return *this;
86  }
87  SourcePosition operator--(int) {
88  SourcePosition tmp(*this);
89  this->operator--();
90  return tmp;
91  }
92 
93  inline const SourcePosition& min(const SourcePosition& rhs) const {
94  if (this->operator<(rhs))
95  return *this;
96  return rhs;
97  }
98 
99  inline const SourcePosition& max(const SourcePosition& rhs) const {
100  if (this->operator>(rhs))
101  return *this;
102  return rhs;
103  }
104 
105  cstring toString() const;
106 
107  bool isValid() const {
108  return this->lineNumber != 0;
109  }
110 
111  unsigned getLineNumber() const {
112  return this->lineNumber;
113  }
114 
115  unsigned getColumnNumber() const {
116  return this->columnNumber;
117  }
118 
119  private:
120  // Input sources where this character position is interpreted.
121  unsigned lineNumber;
122  unsigned columnNumber;
123 };

125 /* Information about the source position of a language element -
126  a range of position within an InputSources. Can only be
127  interpreted relative to some InputSources.
128 
129  For a program element, the start is inclusive and the end is
130  exclusive (the first position after the language element).
131 
132  SourceInfo can also be "invalid"
133 */
134 class SourceInfo final {
135  public:
136  // Creates an "invalid" SourceInfo
137  SourceInfo()
138  : start(SourcePosition()),
139  end(SourcePosition()) {}
140  // Creates a SourceInfo for a 'point' in the source, or invalid
141  explicit SourceInfo(SourcePosition point) : start(point), end(point) {}
142 
143  SourceInfo(SourcePosition start, SourcePosition end);
144 
145  SourceInfo(const SourceInfo& other) = default;
146  ~SourceInfo() = default;
147  SourceInfo& operator=(const SourceInfo& other) = default;
148 
149  // A SourceInfo that spans both this and rhs.
150  // However, if this or rhs is invalid, it is not taken into account
151  const SourceInfo operator+(const SourceInfo& rhs) const {
152  if (!this->isValid())
153  return rhs;
154  if (!rhs.isValid())
155  return *this;
156  SourcePosition s = start.min(rhs.start);
157  SourcePosition e = end.max(rhs.end);
158  return SourceInfo(s, e);
159  }
160  SourceInfo &operator+=(const SourceInfo& rhs) {
161  if (!isValid()) {
162  *this = rhs;
163  } else if (rhs.isValid()) {
164  start = start.min(rhs.start);
165  end = end.max(rhs.end);
166  }
167  return *this;
168  }
169 
170  bool operator==(const SourceInfo &rhs) const
171  { return start == rhs.start && end == rhs.end; }
172 
173  cstring toDebugString() const;
174 
175  void dbprint(std::ostream& out) const
176  { out << this->toDebugString(); }
177 
178  cstring toSourceFragment() const;
179  cstring toPositionString() const;
180  SourceFileLine toPosition() const;
181 
182  bool isValid() const
183  { return this->start.isValid(); }
184  explicit operator bool() const { return isValid(); }
185 
186  const SourcePosition& getStart() const
187  { return this->start; }
188 
189  const SourcePosition& getEnd() const
190  { return this->end; }
191 
192  // True if this comes 'before' this source position
193  // 'invalid' source positions come first.
194  // This is true if the start of other is strictly before
195  // the start of this.
196  bool operator< (const SourceInfo& rhs) const {
197  if (!rhs.isValid()) return false;
198  if (!isValid()) return true;
199  return this->start < rhs.start;
200  }
201  inline bool operator> (const SourceInfo& rhs) const
202  { return rhs.operator< (*this); }
203  inline bool operator<=(const SourceInfo& rhs) const
204  { return !this->operator> (rhs); }
205  inline bool operator>=(const SourceInfo& rhs) const
206  { return !this->operator< (rhs); }
207 
208  private:
209  SourcePosition start;
210  SourcePosition end;
211 };

241  This class implements a singleton pattern: there is a single instance of this class.
242 */
