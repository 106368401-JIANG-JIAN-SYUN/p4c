29 // Determines whether an expression may have side-effects
30 // (but not any sub-expression)
31 class SideEffects : public Inspector {
32  private:
33  ReferenceMap* refMap;
34  TypeMap* typeMap;
35 
36  public:
37  const IR::Node* nodeWithSideEffect = nullptr;
38  unsigned sideEffectCount = 0;
39 
40  void postorder(const IR::MethodCallExpression* mce) override {
41  if (refMap == nullptr || typeMap == nullptr) {
42  // conservative
43  sideEffectCount++;
44  nodeWithSideEffect = mce;
45  return;
46  }
47  // TODO: add an annotation to indicate lack of side-effects
48  auto mi = MethodInstance::resolve(mce, refMap, typeMap);
49  if (!mi->is<BuiltInMethod>()) {
50  sideEffectCount++;
51  nodeWithSideEffect = mce;
52  return;
53  }
54  auto bim = mi->to<BuiltInMethod>();
55  if (bim->name.name != IR::Type_Header::isValid) {
56  sideEffectCount++;
57  nodeWithSideEffect = mce;
58  }
59  }
60  void postorder(const IR::ConstructorCallExpression* cce) override {
61  sideEffectCount++;
62  nodeWithSideEffect = cce;
63  }
64  // If you pass nullptr for these arguments the check will be more conservative
65  SideEffects(ReferenceMap* refMap, TypeMap* typeMap) :
66  refMap(refMap), typeMap(typeMap) { setName("SideEffects"); }
67 
68  // Returns true if the expression may have side-effects.
69  static bool check(const IR::Expression* expression,
70  ReferenceMap* refMap,
71  TypeMap* typeMap) {
72  SideEffects se(refMap, typeMap);
73  expression->apply(se);
74  return se.nodeWithSideEffect != nullptr;
75  }
76 };

78 // convert expresions so that each expression contains at most one side-effect
79 // left-values are converted to contain no side-effects. An important consequence
80 // of this pass is that it converts function calls so that no two parameters
81 // (one of which is output) can alias. This makes the job of the inliner simpler.
82 //
83 // a[f(x)] = b
84 // is converted to
85 // tmp = f(x); a[tmp] = b
86 //
87 // m(n()) is converted to
88 // tmp = n()
89 // m(tmp)
90 //
91 // This is especially tricky for handling out and inout arguments.
92 // Consider bit f(inout T a, in T b); T g(inout T c);
93 // The expression a[g(w)].x = f(a[1].x, g(a[1].x)); is translated as
94 // T tmp1 = g(w); // modifies w
95 // T tmp2 = a[1].x; // save a[1].x
96 // T tmp3 = g(a[1].x); // modifies a[1].x
97 // T tmp4 = f(tmp2, tmp3); // modifies tmp2
98 // a[1].x = tmp2; // copy tmp2 to a[1].x - out argument
99 // a[tmp1].x = tmp4; // assign result of call of f to actual left value
100 // Function call proceeds as follows:
101 // - arguments are evaluated in order
102 // - inout and out arguments produce left-values - these are "saved"
103 // - function is called with temporaries for all arguments
104 // - out and inout temporaries are copied to the saved left-values in order
105 // An assignment statement
106 // e = e1;
107 // is treated as if it is a call to a function "set(out T v, in T v)"
108 // This pass should be run after removing table parameters.
109 // FIXME: does not handle select labels
110 class DoSimplifyExpressions : public Transform {
111  ReferenceMap* refMap;
112  TypeMap* typeMap;
113 
114  IR::IndexedVector<IR::Declaration> toInsert;
115 
116  public:
117  // Currently this only works correctly only if initializers
118  // cannot appear in local declarations.
119  DoSimplifyExpressions(ReferenceMap* refMap, TypeMap* typeMap)
120  : refMap(refMap), typeMap(typeMap) {
121  CHECK_NULL(refMap); CHECK_NULL(typeMap);
122  setName("DoSimplifyExpressions");
123  }
124 
125  const IR::Node* postorder(IR::P4Parser* parser) override;
126  const IR::Node* postorder(IR::Function* function) override;
127  const IR::Node* postorder(IR::P4Control* control) override;
128  const IR::Node* postorder(IR::P4Action* action) override;
129  const IR::Node* postorder(IR::ParserState* state) override;
130  const IR::Node* postorder(IR::AssignmentStatement* statement) override;
131  const IR::Node* postorder(IR::MethodCallStatement* statement) override;
132  const IR::Node* postorder(IR::ReturnStatement* statement) override;
133  const IR::Node* postorder(IR::SwitchStatement* statement) override;
134  const IR::Node* postorder(IR::IfStatement* statement) override;
135 };
