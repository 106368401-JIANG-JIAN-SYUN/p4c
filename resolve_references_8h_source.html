32 class ResolutionContext : public IHasDbPrint {
33  std::vector<const IR::INamespace*> stack;
34  const IR::INamespace* rootNamespace;
35  std::vector<const IR::INamespace*> globals; // match_kind
36 
37  public:
38  explicit ResolutionContext(const IR::INamespace* rootNamespace) :
39  rootNamespace(rootNamespace)
40  { push(rootNamespace); }
41 
42  void dbprint(std::ostream& out) const;
43  void addGlobal(const IR::INamespace* e) {
44  globals.push_back(e);
45  }
46  void push(const IR::INamespace* element) {
47  CHECK_NULL(element);
48  stack.push_back(element);
49  }
50  void pop(const IR::INamespace* element) {
51  if (stack.empty())
52  BUG("Empty stack in ResolutionContext::pop");
53  const IR::INamespace* node = stack.back();
54  if (node != element)
55  BUG("Expected %1% on stack, found %2%", element, node);
56  stack.pop_back();
57  }
58  void done();
