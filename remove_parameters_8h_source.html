26 // For each action that is invoked keep the list of arguments that it's called with.
27 // There must be only one call of each action; this is done by LocalizeActions.
28 class ActionInvocation {
29  std::map<const IR::P4Action*, const IR::MethodCallExpression*> invocations;
30  std::set<const IR::P4Action*> all; // for these actions remove all parameters
31  std::set<const IR::MethodCallExpression*> calls;
32  // how many arguments to remove from each default action
33  std::map<const IR::MethodCallExpression*, unsigned> defaultActions;
34 
35  public:
36  void bind(const IR::P4Action* action, const IR::MethodCallExpression* invocation,
37  bool allParams) {
38  CHECK_NULL(action);
39  BUG_CHECK(invocations.find(action) == invocations.end(),
40  "%1%: action called twice", action);
41  invocations.emplace(action, invocation);
42  if (allParams)
43  all.emplace(action);
44  calls.emplace(invocation);
45  }
46  void bindDefaultAction(const IR::P4Action* action,
47  const IR::MethodCallExpression* defaultInvocation) {
48  // We must have a binding for this action already.
49  auto actionCallInvocation = ::get(invocations, action);
50  CHECK_NULL(actionCallInvocation);
51  // We must remove all arguments which are bound in the action list.
52  unsigned boundArgs = actionCallInvocation->arguments->size();
53  defaultActions.emplace(defaultInvocation, boundArgs);
54  }
55  const IR::MethodCallExpression* get(const IR::P4Action* action) const {
56  return ::get(invocations, action);
57  }
58  bool removeAllParameters(const IR::P4Action* action) const {
59  return all.find(action) != all.end();
60  }
61  bool isCall(const IR::MethodCallExpression* expression) const {
62  return calls.find(expression) != calls.end();
63  }
64  unsigned argsToRemove(const IR::MethodCallExpression* defaultCall) const {
65  if (defaultActions.find(defaultCall) == defaultActions.end())
66  return 0;
67  return ::get(defaultActions, defaultCall);
68  }
69 };

86 // Removes parameters of an action which are in/inout/out.
87 // For this to work each action must have a single caller.
88 // (This is done by the LocalizeActions pass).
89 // control c(inout bit<32> x) {
90 // action a(in bit<32> arg) { x = arg; }
91 // table t() { actions = { a(10); } }
92 // apply { ... } }
93 // is converted to
94 // control c(inout bit<32> x) {
95 // bit<32> arg;
96 // action a() { arg = 10; x = arg; }
97 // table t() { actions = { a; } }
98 // apply { ... } }
99 class DoRemoveActionParameters : public Transform {
100  ActionInvocation* invocations;
101  public:
102  explicit DoRemoveActionParameters(ActionInvocation* invocations) :
103  invocations(invocations)
104  { CHECK_NULL(invocations); setName("DoRemoveActionParameters"); }
105 
106  const IR::Node* postorder(IR::P4Action* table) override;
107  const IR::Node* postorder(IR::ActionListElement* element) override;
108  const IR::Node* postorder(IR::MethodCallExpression* expression) override;
109 };
