28 // A list of equality constraints on types.
29 class TypeConstraints final {
30  class IConstraint {
31  public:
32  virtual ~IConstraint() {}
33  virtual void dbprint(std::ostream& out) const = 0;
34  };
35 
36  class TwoTypeConstraint : public IConstraint {
37  public:
38  const IR::Type* left;
39  const IR::Type* right;
40 
41  protected:
42  TwoTypeConstraint(const IR::Type* left, const IR::Type* right) :
43  left(left), right(right) {
44  CHECK_NULL(left); CHECK_NULL(right);
45  if (left->is<IR::Type_Name>() || right->is<IR::Type_Name>())
46  BUG("Unifying type names %1% and %2%", left, right);
47  }
48  };
49 
50  // Requires two types to be equal.
51  class EqualityConstraint : public TwoTypeConstraint {
52  public:
53  EqualityConstraint(const IR::Type* left, const IR::Type* right)
54  : TwoTypeConstraint(left, right) {}
55  void dbprint(std::ostream& out) const override
56  { out << "Constraint:" << dbp(left) << " = " << dbp(right); }
57  };
58 
59  private:
60  /*
61  * Not all type variables that appear in unification can be bound:
62  * only the ones in this set can be.
63  *
64  * Consider this example:
65  *
66  * extern void f(in bit x);
67  * control p<T>(T data) { f(data); }
68  *
69  * This example should not typecheck: because T cannot be constrained in the invocation of f.
70  * While typechecking the f(data) call, T is not a type variable that can be unified.
71  */
72  std::set<const IR::ITypeVar*> unifiableTypeVariables;
73  std::vector<IConstraint*> constraints;
74  TypeUnification *unification;
75 
76  public:
77  TypeConstraints() : unification(new TypeUnification(this)) {}
78 
79  void addUnifiableTypeVariable(const IR::ITypeVar* typeVariable)
80  { unifiableTypeVariables.insert(typeVariable); }
81 
82  // True if type is a type variable that can be unified.
83  bool isUnifiableTypeVariable(const IR::Type* type) {
84  auto tv = type->to<IR::ITypeVar>();
85  if (tv == nullptr)
86  return false;
87  if (tv->is<IR::Type_InfInt>())
88  // These are always unifiable
89  return true;
90  return unifiableTypeVariables.count(tv) > 0;
91  }
92 
93  void addEqualityConstraint(const IR::Type* left, const IR::Type* right) {
94  auto c = new EqualityConstraint(left, right);
95  LOG1(c);
96  constraints.push_back(c);
97  }
98 
99  /*
100  * Solve the specified constraint.
101  * @param root Element where error is signalled if necessary.
102  * @param subst Variable substitution which is updated with new constraints.
103  * @param constraint Constraint to solve.
104  * @param reportErrors If true report errors.
105  * @return True on success.
106  */
107  bool solve(const IR::Node* root, IConstraint* constraint,
108  TypeVariableSubstitution *subst, bool reportErrors) {
109  auto eq = dynamic_cast<EqualityConstraint*>(constraint);
110  if (eq != nullptr)
111  return solve(root, eq, subst, reportErrors);
112  BUG("unexpected type constraint");
113  }
114 
115  bool solve(const IR::Node* root, EqualityConstraint *constraint,
116  TypeVariableSubstitution *subst, bool reportErrors);
117 
118  TypeVariableSubstitution* solve(const IR::Node* root, bool reportErrors) {
119  LOG1("Solving constraints:\n" << *this);
120 
121  auto tvs = new TypeVariableSubstitution();
122  while (!constraints.empty()) {
123  auto last = constraints.back();
124  constraints.pop_back();
125  bool success = solve(root, last, tvs, reportErrors);
126  if (!success)
127  return nullptr;
128  }
129  LOG1("Constraint solution:\n" << tvs);
130  return tvs;
131  }
132  void dbprint(std::ostream& out) const;
133 };
