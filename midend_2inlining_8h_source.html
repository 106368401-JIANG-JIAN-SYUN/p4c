34 // Describes information about a caller-callee pair
35 struct CallInfo {
36  const IR::IContainer* caller;
37  const IR::IContainer* callee;
38  const IR::Declaration_Instance* instantiation; // callee instantiation
39  // Each instantiation may be invoked multiple times.
40  std::set<const IR::MethodCallStatement*> invocations; // all invocations within the caller
41 
42  CallInfo(const IR::IContainer* caller, const IR::IContainer* callee,
43  const IR::Declaration_Instance* instantiation) :
44  caller(caller), callee(callee), instantiation(instantiation)
45  { CHECK_NULL(caller); CHECK_NULL(callee); CHECK_NULL(instantiation); }
46  void addInvocation(const IR::MethodCallStatement* statement)
47  { invocations.emplace(statement); }
48  void dbprint(std::ostream& out) const
49  { out << "Inline " << callee << " into " << caller; }
50 };

52 class SymRenameMap {
53  std::map<const IR::IDeclaration*, cstring> internalName;
54  std::map<const IR::IDeclaration*, cstring> externalName;
55 
56  public:
57  void setNewName(const IR::IDeclaration* decl, cstring name, cstring extName) {
58  CHECK_NULL(decl);
59  BUG_CHECK(!name.isNullOrEmpty() && !extName.isNullOrEmpty(), "Empty name");
60  LOG1("setNewName " << dbp(decl) << " to " << name);
61  if (internalName.find(decl) != internalName.end())
62  BUG("%1%: already renamed", decl);
63  internalName.emplace(decl, name);
64  externalName.emplace(decl, extName);
65  }
66  cstring getName(const IR::IDeclaration* decl) const {
67  CHECK_NULL(decl);
68  BUG_CHECK(internalName.find(decl) != internalName.end(), "%1%: no new name", decl);
69  auto result = ::get(internalName, decl);
70  return result;
71  }
72  cstring getExtName(const IR::IDeclaration* decl) const {
73  CHECK_NULL(decl);
74  BUG_CHECK(externalName.find(decl) != externalName.end(), "%1%: no external name", decl);
75  auto result = ::get(externalName, decl);
76  return result;
77  }
78  bool isRenamed(const IR::IDeclaration* decl) const {
79  CHECK_NULL(decl);
80  return internalName.find(decl) != internalName.end();
81  }
82 };

97 // Summarizes all inline operations to be performed.
98 struct InlineSummary {
99  // Various substitutions that must be applied for each instance
100  struct PerCaller { // information for each caller
101  // For each instance (key) the container that is intantiated.
102  std::map<const IR::Declaration_Instance*, const IR::IContainer*> declToCallee;
103  // For each instance (key) we must apply a bunch of substitutions
104  std::map<const IR::Declaration_Instance*, PerInstanceSubstitutions*> substitutions;
105  // For each invocation (key) call the instance that is invoked.
106  std::map<const IR::MethodCallStatement*, const IR::Declaration_Instance*> callToInstance;
107  // nullptr if there isn't exactly one caller,
108  // otherwise the single caller of this instance.
109  const IR::MethodCallStatement* uniqueCaller(
110  const IR::Declaration_Instance* instance) const {
111  const IR::MethodCallStatement* call = nullptr;
112  for (auto m : callToInstance) {
113  if (m.second == instance) {
114  if (call == nullptr)
115  call = m.first;
116  else
117  return nullptr;
118  }
119  }
120  return call;
121  }
122  };
123  std::map<const IR::IContainer*, PerCaller> callerToWork;
124 
125  void add(const CallInfo *cci) {
126  callerToWork[cci->caller].declToCallee[cci->instantiation] = cci->callee;
127  for (auto mcs : cci->invocations)
128  callerToWork[cci->caller].callToInstance[mcs] = cci->instantiation;
129  }
130  void dbprint(std::ostream& out) const
131  { out << "Inline " << callerToWork.size() << " call sites"; }
132 };

134 // Inling information constructed here.
135 class InlineWorkList {
136  // We use an ordered map to make the iterator deterministic
137  ordered_map<const IR::Declaration_Instance*, CallInfo*> inlineMap;
138  std::vector<CallInfo*> toInline; // sorted in order of inlining
139 
140  public:
141  void addInstantiation(const IR::IContainer* caller, const IR::IContainer* callee,
142  const IR::Declaration_Instance* instantiation) {
143  CHECK_NULL(caller); CHECK_NULL(callee); CHECK_NULL(instantiation);
144  LOG1("Inline instantiation " << dbp(instantiation));
145  auto inst = new CallInfo(caller, callee, instantiation);
146  inlineMap[instantiation] = inst;
147  }
148 
149  void addInvocation(const IR::Declaration_Instance* instance,
150  const IR::MethodCallStatement* statement) {
151  CHECK_NULL(instance); CHECK_NULL(statement);
152  LOG1("Inline invocation " << dbp(instance) << " at " << dbp(statement));
153  auto info = inlineMap[instance];
154  BUG_CHECK(info, "Could not locate instance %1% invoked by %2%", instance, statement);
155  info->addInvocation(statement);
156  }
157 
158  void replace(const IR::IContainer* container, const IR::IContainer* replacement) {
159  CHECK_NULL(container); CHECK_NULL(replacement);
160  LOG1("Replacing " << dbp(container) << " with " << dbp(replacement));
161  for (auto e : toInline) {
162  if (e->callee == container)
163  e->callee = replacement;
164  if (e->caller == container)
165  e->caller = replacement;
166  }
167  }
168 
169  void analyze(bool allowMultipleCalls);
170  InlineSummary* next();
171 };

182  public:

184  CHECK_NULL(list); CHECK_NULL(toInline);

187  }

190  LOG1("AbstractInliner " << toInline);

262 class Inline : public PassManager {
263  InlineWorkList toInline;
264  public:
265  Inline(ReferenceMap* refMap, TypeMap* typeMap, EvaluatorPass* evaluator) {
266  passes.push_back(new TypeChecking(refMap, typeMap));
267  passes.push_back(new DiscoverInlining(&toInline, refMap, typeMap, evaluator));
268  passes.push_back(new InlineDriver(&toInline, new P4::GeneralInliner(refMap->isV1())));
