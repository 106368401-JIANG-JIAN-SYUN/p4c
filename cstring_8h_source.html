200 template<class T> cstring cstring::make_unique(const T &inuse, cstring base, char sep) {
201  char suffix[8];
202  cstring rv = base;
203  int counter = 0;
204  while (inuse.count(rv)) {
205  snprintf(suffix, sizeof(suffix)/sizeof(suffix[0]), "%c%d", sep, counter++);
206  rv = base + suffix; }
207  return rv; }
