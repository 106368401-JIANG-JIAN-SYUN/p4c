40 // A Vector that also keeps an index of all IDeclaration objects.
41 
42 template<class T>
43 class IndexedVector : public Vector<T> {
44  ordered_map<cstring, const IDeclaration*> declarations;
45 
46  void insertInMap(const T* a) {
47  if (!a->template is<IDeclaration>())
48  return;
49  auto decl = a->template to<IDeclaration>();
50  auto name = decl->getName().name;
51  auto previous = declarations.find(name);
52  if (previous != declarations.end())
53  ::error("%1%: Duplicates declaration %2%", a, previous->second);
54  else
55  declarations[name] = decl; }
56  void removeFromMap(const T* a) {
57  auto decl = a->template to<IDeclaration>();
58  if (decl == nullptr)
59  return;
60  cstring name = decl->getName().name;
61  auto it = declarations.find(name);
62  if (it == declarations.end())
63  BUG("%1% does not exist", a);
64  declarations.erase(it); }
65 
66  public:
67  using Vector<T>::begin;
68  using Vector<T>::end;
69 
70  IndexedVector() = default;
71  IndexedVector(const IndexedVector &) = default;
72  IndexedVector(IndexedVector &&) = default;
73  IndexedVector &operator=(const IndexedVector &) = default;
74  IndexedVector &operator=(IndexedVector &&) = default;
75  explicit IndexedVector(const T *a) {
76  push_back(std::move(a)); }
77  explicit IndexedVector(const vector<const T *> &a) {
78  insert(typename Vector<T>::end(), a.begin(), a.end()); }
79  explicit IndexedVector(const Vector<T> &a) {
80  insert(typename Vector<T>::end(), a.begin(), a.end()); }
81  explicit IndexedVector(JSONLoader &json);
82 
83  void clear() { IR::Vector<T>::clear(); declarations.clear(); }
84  // Although this is not a const_iterator, it should NOT
85  // be used to modify the vector directly. I don't know
86  // how to enforce this property, though.
87  typedef typename Vector<T>::iterator iterator;
88 
89  const IDeclaration* getDeclaration(cstring name) const {
90  auto it = declarations.find(name);
91  if (it == declarations.end())
92  return nullptr;
93  return it->second; }
94  template <class U>
95  const U* getDeclaration(cstring name) const {
96  auto it = declarations.find(name);
97  if (it == declarations.end())
98  return nullptr;
99  return it->second->template to<U>(); }
100  Util::Enumerator<const IDeclaration*>* getDeclarations() const {
101  return Util::Enumerator<const IDeclaration*>::createEnumerator(
102  Values(declarations).begin(), Values(declarations).end()); }
103  iterator erase(iterator i) {
104  removeFromMap(*i);
105  return Vector<T>::erase(i); }
106  template<typename ForwardIter>
107  iterator insert(iterator i, ForwardIter b, ForwardIter e) {
108  for (auto it = b; it != e; ++it)
109  insertInMap(*it);
110  return Vector<T>::insert(i, b, e); }
111  iterator replace(iterator i, const T* v) {
112  removeFromMap(*i);
113  *i = v;
114  insertInMap(v);
115  return ++i; }
116  iterator append(const Vector<T>& toAppend) {
117  return insert(Vector<T>::end(), toAppend.begin(), toAppend.end()); }
118  iterator insert(iterator i, const T* v) {
119  insertInMap(v);
120  return typename Vector<T>::insert(i, v); }
121  template <class... Args> void emplace_back(Args&&... args) {
122  auto el = new T(std::forward<Args>(args)...);
123  insert(el); }
124  void push_back(T *a) { CHECK_NULL(a); Vector<T>::push_back(a); insertInMap(a); }
125  void push_back(const T *a) { CHECK_NULL(a); Vector<T>::push_back(a); insertInMap(a); }
126  void pop_back() {
127  if (typename Vector<T>::empty())
128  BUG("pop_back from empty IndexedVector");
129  auto last = typename Vector<T>::back();
130  removeFromMap(last);
131  typename Vector<T>::pop_back(); }
132  template<class U> void push_back(U &a) { Vector<T>::push_back(a); insertInMap(a); }
