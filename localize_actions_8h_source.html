28 class GlobalActionReplacements {
29  public:
30  // For each control that uses an action and for each each global action
31  // we create a replacement.
32  std::map<const IR::P4Control*, ordered_map<const IR::P4Action*, const IR::P4Action*>*> repl;
33 
34  const IR::P4Action* getReplacement(const IR::P4Action* action,
35  const IR::P4Control* control) const {
36  auto map = ::get(repl, control);
37  if (map == nullptr)
38  return nullptr;
39  if (map->find(action) != map->end())
40  return (*map)[action];
41  return nullptr;
42  }
43  void addReplacement(const IR::P4Action* action, const IR::P4Control* control,
44  const IR::P4Action* replacement) {
45  LOG1("Cloning global " << dbp(action) << " into " <<
46  dbp(replacement) << " for " << dbp(control));
47  if (repl.find(control) == repl.end())
48  repl[control] = new ordered_map<const IR::P4Action*, const IR::P4Action*>();
49  (*repl[control])[action] = replacement;
50  }
51 };

80 class ActionReplacement {
81  public:
82  // For each action and each user the replacement action to use.
83  // Node is either a P4Table or MethodCallExpression.
84  std::map<const IR::P4Action*, ordered_map<const IR::Node*, const IR::P4Action*>*> toInsert;
85  std::map<const IR::PathExpression*, const IR::P4Action*> repl;
86  // For each action all replacements to insert
87 
88  const IR::P4Action* getActionUser(const IR::P4Action* action, const IR::Node* user) {
89  if (toInsert.find(action) == toInsert.end())
90  return nullptr;
91  auto map = toInsert[action];
92  CHECK_NULL(map);
93  if (map->find(user) == map->end())
94  return nullptr;
95  return (*map)[user];
96  }
97  void createReplacement(const IR::P4Action* original, const IR::Node* user,
98  const IR::P4Action* replacement) {
99  auto map = toInsert[original];
100  if (map == nullptr) {
101  map = new ordered_map<const IR::Node*, const IR::P4Action*>();
102  toInsert[original] = map;
103  }
104  (*map)[user] = replacement;
105  }
106  // In the specified path replace the original with the replacement
107  void setRefReplacement(const IR::PathExpression* path,
108  const IR::P4Action* replacement) {
109  LOG1("Adding replacement " << dbp(replacement) << " used by " << dbp(path));
110  repl[path] = replacement;
111  }
112 };

150 class LocalizeAllActions : public PassManager {
151  GlobalActionReplacements globalReplacements;
152  ActionReplacement localReplacements;
153  public:
154  explicit LocalizeAllActions(ReferenceMap* refMap) {
155  passes.emplace_back(new TagGlobalActions());
156  passes.emplace_back(new PassRepeated {
157  new ResolveReferences(refMap),
158  new FindGlobalActionUses(refMap, &globalReplacements),
159  new LocalizeActions(refMap, &globalReplacements),
160  });
161  passes.emplace_back(new ResolveReferences(refMap));
162  passes.emplace_back(new FindRepeatedActionUses(refMap, &localReplacements));
163  passes.emplace_back(new DuplicateActions(&localReplacements));
164  passes.emplace_back(new RemoveAllUnusedDeclarations(refMap));
165  setName("LocalizeAllActions");
166  }
167 };
