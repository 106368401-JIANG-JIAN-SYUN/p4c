32 // This pass only clears the typeMap if the program has changed.
33 // This is needed if the types of some objects in the program change.
34 class ClearTypeMap : public Inspector {
35  TypeMap* typeMap;
36  public:
37  explicit ClearTypeMap(TypeMap* typeMap) :
38  typeMap(typeMap) { CHECK_NULL(typeMap); setName("ClearTypeMap"); }
39  bool preorder(const IR::P4Program* program) override {
40  // Clear map only if program has not changed from last time
41  // otherwise we can reuse it
42  if (!typeMap->checkMap(program))
43  typeMap->clear();
44  return false; // prune()
45  }
46 };

264 // Copy types from the typeMap to expressions. Updates the typeMap with newly created nodes
265 class ApplyTypesToExpressions : public Transform {
266  TypeMap *typeMap;
267  IR::Node *postorder(IR::Node *n) override {
268  const IR::Node *orig = getOriginal();
269  if (auto type = typeMap->getType(orig)) {
270  if (*orig != *n)
271  typeMap->setType(n, type); }
272  return n; }
273  IR::Expression *postorder(IR::Expression *e) override {
274  auto orig = getOriginal<IR::Expression>();
275  if (auto type = typeMap->getType(orig)) {
276  e->type = type;
277  if (*orig != *e) {
278  typeMap->setType(e, type);
279  if (typeMap->isLeftValue(orig))
280  typeMap->setLeftValue(e);
281  if (typeMap->isCompileTimeConstant(orig))
282  typeMap->setCompileTimeConstant(e);
283  }
284  }
285  return e; }
286 
287  public:
288  explicit ApplyTypesToExpressions(TypeMap *typeMap) : typeMap(typeMap)
289  { setName("ApplyTypesToExpressions"); }
290 };
