32 // Information about the structure of a P4-14 program
33 class ProgramStructure {
34  // In P4-14 one can have multiple objects with different types with the same name
35  // In P4-16 this is not possible, so we may need to rename some objects.
36  // We will preserve the original name using an @name("") annotation.
37  template<typename T>
38  class NamedObjectInfo {
39  // If allNames is nullptr we don't check for duplicate names
40  std::unordered_set<cstring> *allNames;
41  std::map<cstring, T> nameToObject;
42  std::map<T, cstring> objectToNewName;
43 
44  // Iterate in order of name, but return pair<T, newname>
45  class iterator {
46  friend class NamedObjectInfo;
47  private:
48  typename std::map<cstring, T>::iterator it;
49  typename std::map<T, cstring> &objToName;
50  iterator(typename std::map<cstring, T>::iterator it,
51  typename std::map<T, cstring> &objToName) :
52  it(it), objToName(objToName) {}
53  public:
54  const iterator& operator++() { ++it; return *this; }
55  bool operator!=(const iterator& other) const { return it != other.it; }
56  std::pair<T, cstring> operator*() const {
57  return std::pair<T, cstring>(it->second, objToName[it->second]);
58  }
59  };
60 
61  public:
62  explicit NamedObjectInfo(std::unordered_set<cstring>* allNames) : allNames(allNames) {}
63  void emplace(T obj) {
64  if (objectToNewName.find(obj) != objectToNewName.end())
65  // Already done
66  return;
67 
68  LOG1("Discovered " << obj);
69  nameToObject.emplace(obj->name, obj);
70  cstring newName;
71 
72  if (allNames == nullptr ||
73  (allNames->find(obj->name) == allNames->end())) {
74  newName = obj->name;
75  } else {
76  newName = cstring::make_unique(*allNames, obj->name, '_');
77  // ::warning("Renaming %1% to %2%", obj, newName);
78  // The old name will be stored in an annotation.
79  }
80  if (allNames != nullptr)
81  allNames->emplace(newName);
82  objectToNewName.emplace(obj, newName);
83  }
84  // Lookup using the original name
85  T get(cstring name) const { return ::get(nameToObject, name); }
86  // Get the new name
87  cstring get(T object) const { return ::get(objectToNewName, object); }
88  bool contains(cstring name) const { return nameToObject.find(name) != nameToObject.end(); }
89  iterator begin() { return iterator(nameToObject.begin(), objectToNewName); }
90  iterator end() { return iterator(nameToObject.end(), objectToNewName); }
91  };
