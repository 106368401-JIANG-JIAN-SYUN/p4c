27 // Given an assignment like
28 // a = lookahead<T>();
29 // this is transformed into
30 // bit<X> tmp = lookahead<sizeof<T>>();
31 // a = { tmp[f1,f0], tmp[f2, f1+1], ... }
32 class DoExpandLookahead : public Transform {
33  P4::ReferenceMap* refMap;
34  P4::TypeMap* typeMap;
35  IR::IndexedVector<IR::Declaration> newDecls;
36 
37  const IR::Expression* expand(
38  const IR::PathExpression* base, const IR::Type* type, unsigned* offset);
39  void expandSetValid(const IR::Expression* base, const IR::Type* type,
40  IR::IndexedVector<IR::StatOrDecl>* output);
41 
42  public:
43  DoExpandLookahead(ReferenceMap* refMap, TypeMap* typeMap) :
44  refMap(refMap), typeMap(typeMap) {
45  CHECK_NULL(refMap); CHECK_NULL(typeMap); setName("DoExpandLookahead"); }
46  const IR::Node* postorder(IR::AssignmentStatement* statement) override;
47  const IR::Node* preorder(IR::P4Control* control) override
48  { prune(); return control; }
49  const IR::Node* preorder(IR::P4Parser* parser) override
50  { newDecls.clear(); return parser; }
51  const IR::Node* postorder(IR::P4Parser* parser) override {
52  if (!newDecls.empty()) {
53  auto locals = new IR::IndexedVector<IR::Declaration>(*parser->parserLocals);
54  locals->append(newDecls);
55  parser->parserLocals = locals;
56  }
57  return parser;
58  }
59 };
