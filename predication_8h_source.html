23 // Converts if statements to ?: in actions, if possible
24 // For this to work all statements must be assignments or other ifs.
25 namespace P4 {
26 
27 /*
28 
29 if (e)
30  a = f(b);
31 else
32  c = f(d);
33 x = y;
34 
35 becomes (actual implementatation is slightly optimized):
36 
37 bool cond;
38 bool predicate = true;
39 {
40  bool predicate1;
41  cond = e;
42  predicate1 = predicate && cond;
43  a = predicate1 ? f(b) : a;
44  cond = !cond;
45  predicate1 = predicate && cond;
46  c = predicate ? f(d) : c;
47 }
48 x = predicate ? y : x;
49 
50 Not the most efficient conversion currently.
51 This could be made better by looking on both the "then" and "else"
52 branches, but in this way we cannot have two side-effects in the same
53 conditional statement.
54 
55 */
56 class Predication final : public Transform {
57  NameGenerator* generator;
58  bool inside_action;
59  std::vector<cstring> predicateName;
60  unsigned ifNestingLevel;
61 
62  const IR::Expression* predicate() const {
63  if (predicateName.empty())
64  return nullptr;
65  return new IR::PathExpression(IR::ID(predicateName.back())); }
66  const IR::Statement* error(const IR::Statement* statement) const {
67  if (inside_action && ifNestingLevel > 0)
68  ::error("%1%: Predication cannot be applied", statement);
69  return statement;
70  }
71 
72  public:
73  explicit Predication(NameGenerator* generator) : generator(generator),
74  inside_action(false), ifNestingLevel(0)
75  { CHECK_NULL(generator); setName("Predication"); }
76 
77  const IR::Node* preorder(IR::IfStatement* statement) override;
78  const IR::Node* preorder(IR::P4Action* action) override;
79  const IR::Node* postorder(IR::P4Action* action) override;
80  const IR::Node* postorder(IR::AssignmentStatement* statement) override;
81  // The presence of other statements makes predication impossible to apply
82  const IR::Node* postorder(IR::MethodCallStatement* statement) override
83  { return error(statement); }
84  const IR::Node* postorder(IR::ReturnStatement* statement) override
85  { return error(statement); }
86  const IR::Node* postorder(IR::ExitStatement* statement) override
87  { return error(statement); }
88 };

90 } // namespace P4
