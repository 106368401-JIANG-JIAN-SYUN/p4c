95 class ConstantFolding : public PassManager {
96  public:
97  ConstantFolding(ReferenceMap* refMap, TypeMap* typeMap, bool warnings = true) {
98  if (typeMap != nullptr)
99  passes.push_back(new TypeChecking(refMap, typeMap));
100  passes.push_back(new DoConstantFolding(refMap, typeMap, warnings));
101  setName("ConstantFolding");
102  }
103 };
