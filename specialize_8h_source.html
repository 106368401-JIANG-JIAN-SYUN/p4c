27 // Specializes Parsers and Controls by substituting type arguments
28 // and constructor parameters.
29 // control c<T>(in T data)(bit<32> size) { ... }
30 // c<bit<32>>(16) c_inst;
31 // is converted to
32 // control cspec(in bit<32> data) { ... }
33 // cspec() c_inst;
34 // A different specialization is made for each constructor invocation
35 // and Declaration_Instance.
36 
37 // Describes how a parser or control is specialized:
38 struct SpecializationInfo {
39  // Name to use for specialized object
40  cstring name;
41  // Actual parser or control that is being specialized
42  const IR::IContainer* specialized;
43  // values to substitute for type arguments
44  const IR::Vector<IR::Type>* typeArguments;
45  // values to substitute for constructor arguments
46  IR::Vector<IR::Expression>* constructorArguments;
47  // declarations to insert in the list of locals
48  IR::IndexedVector<IR::Declaration>* declarations;
49  // Invocation which causes this specialization
50  const IR::Node* invocation;
51  // Where in the program should the specialization be inserted
52  const IR::Node* insertBefore;
53 
54  SpecializationInfo(const IR::Node* invocation, const IR::IContainer* cont,
55  const IR::Node* insertion) :
56  specialized(cont), typeArguments(nullptr),
57  constructorArguments(new IR::Vector<IR::Expression>()),
58  declarations(new IR::IndexedVector<IR::Declaration>()),
59  invocation(invocation), insertBefore(insertion)
60  { CHECK_NULL(cont); CHECK_NULL(invocation); CHECK_NULL(insertion); }
61  const IR::Type_Declaration* synthesize(ReferenceMap* refMap) const;
62 };

68 
69  public:
70  TypeMap* typeMap;

72  void addSpecialization(
73  const IR::ConstructorCallExpression* invocation,
74  const IR::IContainer* container, // actual object specialized
75  const IR::Node* insertion); // where the specialization should be inserted
76  void addSpecialization(
77  const IR::Declaration_Instance* invocation,
78  const IR::IContainer* container, // actual object specialized
79  const IR::Node* insertion); // where the specialization should be inserted
80  IR::Vector<IR::Node>* getSpecializations(const IR::Node* insertion) const;
81  cstring getName(const IR::Node* insertion) const {
82  auto s = ::get(specializations, insertion);
83  if (s == nullptr)
84  return nullptr;
85  return s->name;
86  }
87  void clear() { specializations.clear(); }
